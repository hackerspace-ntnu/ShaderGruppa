cmake_minimum_required(VERSION 3.20)
project(ShaderGruppa LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

# Don't build GLFW's docs/tests/examples (faster, avoids warning spam)
set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(GLFW_INSTALL OFF CACHE BOOL "" FORCE)

FetchContent_Declare(
  glfw
  GIT_REPOSITORY https://github.com/glfw/glfw.git
  GIT_TAG 3.3.9
)
FetchContent_MakeAvailable(glfw)

# glad (vendored in repo)
if (EXISTS "${CMAKE_SOURCE_DIR}/external/glad/src/glad.c")
  add_library(glad STATIC external/glad/src/glad.c)
elseif (EXISTS "${CMAKE_SOURCE_DIR}/external/glad/src/gl.c")
  add_library(glad STATIC external/glad/src/gl.c)
else()
  message(FATAL_ERROR "Could not find glad source file in external/glad/src (expected glad.c or gl.c)")
endif()

target_include_directories(glad PUBLIC external/glad/include)

add_executable(shadergruppa
  src/main.cpp
  src/file_io.cpp
  src/gl_shader.cpp
  src/gl_texture.cpp
  src/pingpong.cpp
  src/menu.cpp
  src/gl_png_to_texture.cpp
  external/lodepng/lodepng.cpp
)

# Allow #include "lodepng.h"
target_include_directories(shadergruppa PRIVATE external/lodepng)

set_target_properties(shadergruppa PROPERTIES
    OUTPUT_NAME "DiffusionSim"
)

target_link_libraries(shadergruppa PRIVATE glfw glad)

if (WIN32)
  target_link_libraries(shadergruppa PRIVATE opengl32)
elseif(APPLE)
  target_link_libraries(shadergruppa PRIVATE "-framework OpenGL" "-framework Cocoa" "-framework IOKit" "-framework CoreVideo")
else()
  target_link_libraries(shadergruppa PRIVATE GL dl pthread)
endif()

# Copy shaders next to the binary so relative paths work
add_custom_command(TARGET shadergruppa POST_BUILD
  COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:shadergruppa>/shaders
  COMMAND ${CMAKE_COMMAND} -E copy_directory
          ${CMAKE_SOURCE_DIR}/shaders
          $<TARGET_FILE_DIR:shadergruppa>/shaders
)
